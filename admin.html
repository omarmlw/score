⚙️ إعداد Google Apps Script (مرة واحدة فقط)

لكي يتمكن الأدمن من إضافة/تعديل/حذف المباريات، تحتاج نشر Apps Script كـ Web App. اتبع الخطوات:

1
افتح الـ Google Sheet الخاص بك، ثم اذهب إلى: Extensions → Apps Script
2
احذف الكود الموجود والصق هذا الكود كاملاً:
const SHEET_ID = '1eG9XRAHv8Fn-tkFmEsbN1xPg7j7bi_4xo69rOyw5WPw'; const SHEET_NAME = 'Sheet1'; function doPost(e) { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); const action = data.action; let result = { success: true }; try { if (action === 'add') { const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const row = headers.map(h => data.row[h] || ''); sheet.appendRow(row); } else if (action === 'update') { const rowNum = parseInt(data.rowNum) + 2; // +2: header + 1-based const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; headers.forEach((h, i) => { if (data.row[h] !== undefined) { sheet.getRange(rowNum, i + 1).setValue(data.row[h]); } }); } else if (action === 'delete') { const rowNum = parseInt(data.rowNum) + 2; sheet.deleteRow(rowNum); } else if (action === 'bulkUpdate') { // update a single field for all rows const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const colIdx = headers.indexOf(data.field) + 1; if (colIdx > 0) { const lastRow = sheet.getLastRow(); for (let r = 2; r <= lastRow; r++) { sheet.getRange(r, colIdx).setValue(data.value); } } } } catch(err) { result = { success: false, error: err.message }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); }
3
اضغط Deploy → New Deployment → اختر Web App → في "Who has access" اختر Anyone → اضغط Deploy
4
